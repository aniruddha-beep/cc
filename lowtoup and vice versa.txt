%{
#include <stdio.h>

/* Manual case conversion without using ctype.h */
char convert(char c) {
    if (c >= 'a' && c <= 'z') {
        c = c - 32;  // lowercase → uppercase
    } else if (c >= 'A' && c <= 'Z') {
        c = c + 32;  // uppercase → lowercase
    }
    return c;
}
%}

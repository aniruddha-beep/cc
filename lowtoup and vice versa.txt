%{
#include <stdio.h>
#include <ctype.h>
%}

%%
[A-Z]     { printf("%c", tolower(yytext[0])); }   // Convert uppercase → lowercase
[a-z]     { printf("%c", toupper(yytext[0])); }   // Convert lowercase → uppercase
.|\n      { printf("%c", yytext[0]); }            // Print other characters as-is
%%

int main() {
    printf("Enter a string: ");
    yylex();   // Start scanning user input
    printf("\n");
    return 0;
}

int yywrap() { return 1; }
